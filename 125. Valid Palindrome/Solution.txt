class Solution {
    public boolean isPalindrome(String s) {
        if(s.length()<=1) return true;
        s=s.toLowerCase();
        int left=0;
        int right=s.length()-1;
        while(left<right){
            char a= s.charAt(left);
            char b=s.charAt(right);
            if (!((a >= 'a' && a <= 'z') || (a >= '0' && a <= '9'))) {
                left++;
                continue;
            }
            if (!((b >= 'a' && b <= 'z') || (b >= '0' && b <= '9'))) {
                right--;
                continue;
            }

     
            if (a != b) return false;

            left++;
            right--;
        }
        return true;
    }
}
